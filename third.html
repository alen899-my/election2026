Backend — Node.js + Express

Express Server

The Kerala 2026 backend is a modular Express.js server written in TypeScript. It follows a feature-module pattern — each domain (constituencies, candidates, parties, results) lives in its own folder with its own routes, controller, service, and model. The server runs on Railway and exposes a versioned REST API consumed by the Next.js frontend.

Server Entry Point

TypeScript — apps/api/src/index.ts
copy
import express from 'express';
import cors from 'cors';
import helmet from 'helmet';
import morgan from 'morgan';
import compression from 'compression';
import { rateLimit } from 'express-rate-limit';
import { env } from './config/env';
import { errorHandler } from './middleware/errorHandler';
import { notFoundHandler } from './middleware/notFoundHandler';
import constituenciesRouter from './modules/constituencies/constituencies.routes';
import candidatesRouter from './modules/candidates/candidates.routes';
import partiesRouter from './modules/parties/parties.routes';
import resultsRouter from './modules/results/results.routes';
import searchRouter from './modules/search/search.routes';
import uploadRouter from './modules/upload/upload.routes';
import statsRouter from './modules/stats/stats.routes';

const app = express();

/* ── Security ─────────────────────────────── */
app.use(helmet());
app.use(cors({
  origin: env.ALLOWED_ORIGINS.split(','),
  methods: ['GET', 'POST', 'PUT', 'DELETE'],
  credentials: true,
}));

/* ── Rate Limiting ────────────────────────── */
const limiter = rateLimit({
  windowMs: env.RATE_LIMIT_WINDOW_MS,    // 15 minutes
  max:      env.RATE_LIMIT_MAX,           // 100 requests
  standardHeaders: true,
  legacyHeaders:   false,
  message: { success: false, error: 'Too many requests. Please slow down.' },
});
app.use('/api', limiter);

/* ── Body Parsing & Compression ──────────── */
app.use(compression());
app.use(express.json({ limit: '10mb' }));
app.use(express.urlencoded({ extended: true }));

/* ── Logging ──────────────────────────────── */
app.use(morgan(env.NODE_ENV === 'production' ? 'combined' : 'dev'));

/* ── Health Check ─────────────────────────── */
app.get('/health', (_, res) => {
  res.json({ status: 'ok', version: '1.0.0', uptime: process.uptime() });
});

/* ── API Routes ───────────────────────────── */
app.use('/api/v1/constituencies', constituenciesRouter);
app.use('/api/v1/candidates',    candidatesRouter);
app.use('/api/v1/parties',       partiesRouter);
app.use('/api/v1/results',       resultsRouter);
app.use('/api/v1/search',        searchRouter);
app.use('/api/v1/upload',        uploadRouter);
app.use('/api/v1/stats',         statsRouter);

/* ── Error Handlers ───────────────────────── */
app.use(notFoundHandler);
app.use(errorHandler);

app.listen(env.PORT, () => {
  console.log(`Kerala 2026 API running on http://localhost:${env.PORT}`);
});

export default app;

Module Structure Pattern

Every feature module follows the same four-file pattern: routes (URL definitions), controller (request/response handling), service (business logic + DB queries), and an optional model (Drizzle schema helpers). This keeps concerns cleanly separated.

├── candidates.routes.ts      # Express Router — URL definitions only
├── candidates.controller.ts  # req/res, parse params, call service, return JSON
├── candidates.service.ts     # All business logic + Drizzle ORM queries
├── candidates.schema.ts      # Zod validation schemas for request bodies

Express Middleware Stack

Middleware

Kerala 2026 uses a layered middleware stack. Security headers (Helmet), CORS, rate limiting, and compression are global. Validation, authentication, and error handling are applied per-route where needed.

Global Middleware Order

1
helmet()
Sets security HTTP headers: X-Content-Type-Options, X-Frame-Options, Strict-Transport-Security, Content-Security-Policy, and more. Applied to every request.
2
cors()
Restricts cross-origin requests to origins listed in ALLOWED_ORIGINS env var. In production: https://kerala2026.in and https://admin.kerala2026.in only.
3
rateLimit() — /api/* only
100 requests per 15-minute window per IP. Returns 429 with JSON error on breach. Uses standard headers (RateLimit-Remaining, RateLimit-Reset).
4
compression()
Gzip/deflate compression for all responses over 1KB. Critical for large constituency GeoJSON payloads and candidate list responses.
5
express.json() + urlencoded()
Body parsing limited to 10MB to support GeoJSON uploads in the admin panel. All incoming bodies are validated by Zod schemas in the route handlers.
6
morgan() — logging
Development: colored dev format. Production: combined Apache format piped to stdout, collected by Railway logging infrastructure.
7
notFoundHandler → errorHandler
Catch-all 404 handler, then global error handler. Both return consistent JSON envelope format matching the API response contract defined in Part 2.

Error Handler

TypeScript — middleware/errorHandler.ts
copy
import { Request, Response, NextFunction } from 'express';
import { ZodError } from 'zod';

export class AppError extends Error {
  constructor(
    public message: string,
    public statusCode: number = 500,
    public code: string = 'INTERNAL_ERROR'
  ) {
    super(message);
    this.name = 'AppError';
  }
}

export function errorHandler(
  err: Error,
  _req: Request,
  res: Response,
  _next: NextFunction
) {
  // Zod validation error
  if (err instanceof ZodError) {
    return res.status(422).json({
      success: false,
      error:   'Validation failed',
      code:    'VALIDATION_ERROR',
      issues:  err.flatten().fieldErrors,
    });
  }

  // Known application error
  if (err instanceof AppError) {
    return res.status(err.statusCode).json({
      success: false,
      error:   err.message,
      code:    err.code,
    });
  }

  // Drizzle / Postgres constraint violations
  if ((err as any).code === '23505') {
    return res.status(409).json({
      success: false,
      error:   'A record with this value already exists',
      code:    'DUPLICATE_ENTRY',
    });
  }

  // Unknown / unexpected error — log and return generic message
  console.error('Unhandled error:', err);
  res.status(500).json({
    success: false,
    error:   'Internal server error',
    code:    'INTERNAL_ERROR',
  });
}

export function notFoundHandler(_req: Request, res: Response) {
  res.status(404).json({
    success: false,
    error:   'Route not found',
    code:    'NOT_FOUND',
  });
}

Validation Middleware

TypeScript — middleware/validate.ts
copy
import { Request, Response, NextFunction } from 'express';
import { ZodSchema } from 'zod';

// Validates req.body against a Zod schema. Calls next(err) on failure.
export function validateBody(schema: ZodSchema) {
  return (req: Request, _res: Response, next: NextFunction) => {
    const result = schema.safeParse(req.body);
    if (!result.success) return next(result.error);
    req.body = result.data;  // replace with parsed/coerced data
    next();
  };
}

// Validates req.query
export function validateQuery(schema: ZodSchema) {
  return (req: Request, _res: Response, next: NextFunction) => {
    const result = schema.safeParse(req.query);
    if (!result.success) return next(result.error);
    (req as any).parsedQuery = result.data;
    next();
  };
}

Authentication

Auth & Admin

The public API requires no authentication — all GET endpoints are open. Write operations (POST, PUT, DELETE) are restricted to admin users authenticated via JWT. There is a single admin account; this is an internal data-entry tool, not a multi-user system.

Design Decision: A single admin JWT keeps the system simple. Only a small trusted team enters candidate data. If multiple coordinators are needed, extend the admin_users table with individual logins.

Admin Login Flow

TypeScript — modules/auth/auth.service.ts
copy
import bcrypt from 'bcryptjs';
import jwt from 'jsonwebtoken';
import { env } from '../../config/env';
import { AppError } from '../../middleware/errorHandler';

export async function adminLogin(password: string) {
  const valid = await bcrypt.compare(password, env.ADMIN_PASSWORD_HASH);
  if (!valid) {
    throw new AppError('Invalid credentials', 401, 'UNAUTHORIZED');
  }

  const token = jwt.sign(
    { role: 'admin', iat: Date.now() },
    env.ADMIN_JWT_SECRET,
    { expiresIn: '12h' }
  );

  return { token, expiresIn: '12h' };
}

// Generate the ADMIN_PASSWORD_HASH for your .env:
// node -e "require('bcryptjs').hash('your_password', 12).then(console.log)"

JWT Auth Middleware

TypeScript — middleware/auth.ts
copy
import { Request, Response, NextFunction } from 'express';
import jwt from 'jsonwebtoken';
import { env } from '../config/env';
import { AppError } from './errorHandler';

export function requireAdmin(
  req: Request,
  _res: Response,
  next: NextFunction
) {
  const authHeader = req.headers.authorization;
  if (!authHeader?.startsWith('Bearer ')) {
    throw new AppError('No token provided', 401, 'UNAUTHORIZED');
  }

  const token = authHeader.slice(7);
  try {
    const payload = jwt.verify(token, env.ADMIN_JWT_SECRET) as { role: string };
    if (payload.role !== 'admin') {
      throw new AppError('Forbidden', 403, 'FORBIDDEN');
    }
    next();
  } catch (e) {
    if (e instanceof jwt.TokenExpiredError) {
      throw new AppError('Token expired', 401, 'TOKEN_EXPIRED');
    }
    if (e instanceof AppError) throw e;
    throw new AppError('Invalid token', 401, 'UNAUTHORIZED');
  }
}

// Usage in routes:
// router.post('/', requireAdmin, validateBody(schema), controller.create)

Cloudflare R2 — Media Storage

File Uploads

Candidate photos and party logos are stored in Cloudflare R2 — an S3-compatible object store with zero egress fees. Photos are automatically resized to 600×800px WebP on upload using the Sharp library. CDN-delivered globally via Cloudflare's edge network.

Why Cloudflare R2? With 1000+ candidate photos at ~200KB each after compression, egress costs on AWS S3 would add up. R2's zero egress pricing makes it ideal for an open-source project at this scale.

Photo Upload Service

TypeScript — modules/upload/upload.service.ts
copy
import sharp from 'sharp';
import { PutObjectCommand } from '@aws-sdk/client-s3';
import { randomUUID } from 'crypto';
import { r2Client } from '../../config/r2';
import { env } from '../../config/env';

const PHOTO_WIDTH  = 600;
const PHOTO_HEIGHT = 800;
const MAX_SIZE_BYTES = 5 * 1024 * 1024; // 5MB input limit

export async function uploadCandidatePhoto(
  buffer:       Buffer,
  originalName: string,
  candidateSlug: string
): Promise<string> {
  if (buffer.byteLength > MAX_SIZE_BYTES) {
    throw new Error('File too large. Max 5MB.');
  }

  // Resize + convert to WebP using Sharp
  const optimised = await sharp(buffer)
    .resize(PHOTO_WIDTH, PHOTO_HEIGHT, {
      fit:      'cover',
      position: 'top',   // keep face, crop bottom
    })
    .webp({ quality: 85 })
    .toBuffer();

  const key = `candidates/${candidateSlug}/photo-${randomUUID()}.webp`;

  await r2Client.send(new PutObjectCommand({
    Bucket:      env.CF_R2_BUCKET,
    Key:         key,
    Body:        optimised,
    ContentType: 'image/webp',
    CacheControl: 'public, max-age=31536000, immutable',
    Metadata: {
      candidateSlug,
      originalName,
    },
  }));

  // Return public CDN URL
  return `${env.CF_R2_PUBLIC_URL}/${key}`;
}

export async function uploadPartyLogo(buffer: Buffer, partySlug: string) {
  const optimised = await sharp(buffer)
    .resize(200, 200, { fit: 'contain', background: { r: 0, g: 0, b: 0, alpha: 0 } })
    .webp({ quality: 90 })
    .toBuffer();

  const key = `parties/${partySlug}/logo.webp`;

  await r2Client.send(new PutObjectCommand({
    Bucket:      env.CF_R2_BUCKET,
    Key:         key,
    Body:        optimised,
    ContentType: 'image/webp',
    CacheControl: 'public, max-age=31536000, immutable',
  }));

  return `${env.CF_R2_PUBLIC_URL}/${key}`;
}

Upload Route

TypeScript — modules/upload/upload.routes.ts
copy
import { Router } from 'express';
import multer from 'multer';
import { requireAdmin } from '../../middleware/auth';
import { uploadCandidatePhoto } from './upload.service';

const router = Router();

// Memory storage — file lives in RAM during processing
const upload = multer({
  storage: multer.memoryStorage(),
  limits:  { fileSize: 5 * 1024 * 1024 },
  fileFilter: (_req, file, cb) => {
    const allowed = ['image/jpeg', 'image/png', 'image/webp'];
    cb(null, allowed.includes(file.mimetype));
  },
});

router.post(
  '/candidate-photo',
  requireAdmin,
  upload.single('photo'),
  async (req, res, next) => {
    try {
      if (!req.file)  throw new Error('No file uploaded');
      const { candidateSlug } = req.body;
      const url = await uploadCandidatePhoto(
        req.file.buffer,
        req.file.originalname,
        candidateSlug
      );
      res.json({ success: true, data: { url } });
    } catch (e) { next(e); }
  }
);

export default router;

Database — PostgreSQL (Neon) + Drizzle ORM

Drizzle ORM

Kerala 2026 uses Drizzle ORM for all database access. Drizzle is a lightweight, fully type-safe SQL toolkit that feels like writing SQL but with TypeScript autocompletion. It is the only ORM that works seamlessly in both Node.js and edge runtimes, which matters for Next.js API routes.

Drizzle vs Prisma: We chose Drizzle because it produces a smaller bundle, has better TypeScript inference, works natively in Next.js edge runtimes, and gives direct SQL-like control without "magic". The trade-off is slightly more verbose queries for complex joins.

import { db } from '../config/db';
import { districts, constituencies, parties } from '../schema';
import { DISTRICTS_DATA } from './data/districts';
import { CONSTITUENCIES_DATA } from './data/constituencies';
import { PARTIES_DATA } from './data/parties';

async function seed() {
  console.log('🌱 Seeding Kerala 2026 database...');

  // 1. Insert 14 districts
  console.log('  → Seeding districts...');
  const insertedDistricts = await db
    .insert(districts)
    .values(DISTRICTS_DATA)
    .onConflictDoNothing()
    .returning();

  // Build slug → id map for FK references
  const districtMap = new Map(insertedDistricts.map(d => [d.slug, d.id]));

  // 2. Insert 140 constituencies with correct district FK
  console.log('  → Seeding 140 constituencies...');
  await db
    .insert(constituencies)
    .values(CONSTITUENCIES_DATA.map(c => ({
      ...c,
      districtId: districtMap.get(c.districtSlug)!,
    })))
    .onConflictDoNothing();

  // 3. Insert parties (LDF/UDF/NDA components)
  console.log('  → Seeding parties...');
  await db
    .insert(parties)
    .values(PARTIES_DATA)
    .onConflictDoNothing();

  console.log('✅ Seed complete!');
  console.log(`   Districts:     ${DISTRICTS_DATA.length}`);
  console.log(`   Constituencies: ${CONSTITUENCIES_DATA.length}`);
  console.log(`   Parties:        ${PARTIES_DATA.length}`);
  process.exit(0);
}

seed().catch(e => { console.error(e); process.exit(1); });

Paginated List with Filters

TypeScript — modules/candidates/candidates.service.ts

import { eq, and, ilike, gt, isNull, desc, asc, sql } from 'drizzle-orm';
import { db } from '../../config/db';
import { candidates, parties, constituencies, districts } from '@kerala2026/db';
import type { ListCandidatesQuery } from './candidates.schema';

export async function listCandidates(query: ListCandidatesQuery) {
  const { page, limit, q, party, district, gender, isIncumbent, hasCriminalCases, sort } = query;
  const offset = (page - 1) * limit;

  // Build dynamic WHERE conditions
  const conditions = [isNull(candidates.deletedAt), eq(candidates.electionYear, 2026)];

  if (q)             conditions.push(ilike(candidates.nameEn, `%${q}%`));
  if (gender)        conditions.push(eq(candidates.gender, gender));
  if (isIncumbent)   conditions.push(eq(candidates.isIncumbent, isIncumbent === 'true'));
  if (hasCriminalCases === 'true') conditions.push(gt(candidates.criminalCases, 0));

  // Sort mapping
  const sortMap = {
    name:   asc(candidates.nameEn),
    assets: desc(candidates.totalAssetsInr),
    terms:  desc(candidates.termsServed),
  };

  const [rows, countResult] = await Promise.all([
    db.select({
      id:           candidates.id,
      slug:         candidates.slug,
      nameEn:       candidates.nameEn,
      nameMl:       candidates.nameMl,
      photoUrl:     candidates.photoUrl,
      isIncumbent:  candidates.isIncumbent,
      termsServed:  candidates.termsServed,
      criminalCases: candidates.criminalCases,
      totalAssetsInr: candidates.totalAssetsInr,
      gender:       candidates.gender,
      nominationStatus: candidates.nominationStatus,
      party: {
        id:           parties.id,
        slug:         parties.slug,
        nameEn:       parties.nameEn,
        abbreviation: parties.abbreviation,
        coalition:    parties.coalition,
        colorHex:     parties.colorHex,
      },
      constituency: {
        id:     constituencies.id,
        slug:   constituencies.slug,
        nameEn: constituencies.nameEn,
        nameMl: constituencies.nameMl,
        number: constituencies.number,
      },
    })
    .from(candidates)
    .leftJoin(parties, eq(candidates.partyId, parties.id))
    .leftJoin(constituencies, eq(candidates.constituencyId, constituencies.id))
    .leftJoin(districts, eq(constituencies.districtId, districts.id))
    .where(and(...conditions))
    .orderBy(sortMap[sort])
    .limit(limit)
    .offset(offset),

    db.select({ count: sql`count(*)` })
    .from(candidates)
    .leftJoin(constituencies, eq(candidates.constituencyId, constituencies.id))
    .leftJoin(districts, eq(constituencies.districtId, districts.id))
    .where(and(...conditions)),
  ]);

  const total      = Number(countResult[0]?.count ?? 0);
  const totalPages = Math.ceil(total / limit);

  return { data: rows, pagination: { page, limit, total, totalPages } };
}

import { candidateAssets, criminalCases, electionResults } from '@kerala2026/db';

export async function getCandidateBySlug(slug: string) {
  // Base candidate with party + constituency
  const [row] = await db
    .select()
    .from(candidates)
    .leftJoin(parties, eq(candidates.partyId, parties.id))
    .leftJoin(constituencies, eq(candidates.constituencyId, constituencies.id))
    .leftJoin(districts, eq(constituencies.districtId, districts.id))
    .where(and(
      eq(candidates.slug, slug),
      isNull(candidates.deletedAt)
    ))
    .limit(1);

  if (!row) return null;

  const candidateId = row.candidates.id;

  // Parallel fetch: assets + criminal cases + election history
  const [assets, cases, history] = await Promise.all([
    db.select().from(candidateAssets)
      .where(eq(candidateAssets.candidateId, candidateId)),

    db.select().from(criminalCases)
      .where(eq(criminalCases.candidateId, candidateId)),

    db.select({
        year:          electionResults.electionYear,
        constituencyName: constituencies.nameEn,
        partyName:     parties.nameEn,
        votesReceived: electionResults.votesReceived,
        position:      electionResults.position,
        isWinner:      electionResults.isWinner,
        winningMargin: electionResults.winningMargin,
      })
      .from(electionResults)
      .leftJoin(constituencies, eq(electionResults.constituencyId, constituencies.id))
      .leftJoin(parties, eq(electionResults.partyId, parties.id))
      .where(eq(electionResults.candidateId, candidateId))
      .orderBy(desc(electionResults.electionYear)),
  ]);

  return {
    ...row.candidates,
    party:        row.parties,
    constituency: { ...row.constituencies, district: row.districts },
    assets,
    criminalCases: cases,
    electionHistory: history,
  };
}

Admin Middleware (Next.js)

TypeScript — apps/admin/middleware.ts

import { NextRequest, NextResponse } from 'next/server';
import { jwtVerify } from 'jose';

const PUBLIC_PATHS = ['/login', '/api/auth/login'];

export async function middleware(req: NextRequest) {
  const { pathname } = req.nextUrl;

  if (PUBLIC_PATHS.some(p => pathname.startsWith(p))) {
    return NextResponse.next();
  }

  const token = req.cookies.get('admin_token')?.value;
  if (!token) {
    return NextResponse.redirect(new URL('/login', req.url));
  }

  try {
    const secret = new TextEncoder().encode(process.env.ADMIN_JWT_SECRET);
    await jwtVerify(token, secret);
    return NextResponse.next();
  } catch {
    const res = NextResponse.redirect(new URL('/login', req.url));
    res.cookies.delete('admin_token');

import Papa from 'papaparse';
import { z } from 'zod';
import { db } from '../../config/db';
import { candidates, constituencies, parties } from '@kerala2026/db';
import { eq } from 'drizzle-orm';

const csvRowSchema = z.object({
  name_en:            z.string().min(2),
  name_ml:            z.string().optional(),
  constituency_slug:  z.string(),
  party_slug:         z.string(),
  dob:                z.string().regex(/^\d{4}-\d{2}-\d{2}$/).optional(),
  gender:             z.enum(['male','female','other']).optional(),
  education:          z.string().optional(),
  profession:         z.string().optional(),
  is_incumbent:       z.enum(['true','false']).default('false'),
  terms_served:       z.coerce.number().int().min(0).default(0),
  criminal_cases:     z.coerce.number().int().min(0).default(0),
  total_assets_inr:   z.coerce.number().int().optional(),
  total_liabilities_inr: z.coerce.number().int().optional(),
  affidavit_url:      z.string().url().optional().or(z.literal('')),
  nomination_status:  z.enum(['filed','accepted','withdrawn','rejected']).default('filed'),
});

export async function importCandidatesFromCSV(csvText: string) {
  const { data } = Papa.parse(csvText, { header: true, skipEmptyLines: true });

  // Pre-load lookup maps
  const [allConstituencies, allParties] = await Promise.all([
    db.select({ id: constituencies.id, slug: constituencies.slug }).from(constituencies),
    db.select({ id: parties.id, slug: parties.slug }).from(parties),
  ]);

  const constMap  = new Map(allConstituencies.map(c => [c.slug, c.id]));
  const partyMap  = new Map(allParties.map(p => [p.slug, p.id]));

  const errors:  Array<{ row: number, issues: any }> = [];
  const toInsert: any[] = [];

  (data as any[]).forEach((row, i) => {
    const parsed = csvRowSchema.safeParse(row);
    if (!parsed.success) {
      errors.push({ row: i + 2, issues: parsed.error.flatten().fieldErrors });
      return;
    }

    const r = parsed.data;
    const constituencyId = constMap.get(r.constituency_slug);
    const partyId        = partyMap.get(r.party_slug);

    if (!constituencyId) {
      errors.push({ row: i + 2, issues: { constituency_slug: 'Unknown constituency slug' } });
      return;
    }

    toInsert.push({
      constituencyId,
      partyId: partyId || null,
      nameEn:   r.name_en,
      nameMl:   r.name_ml,
      slug:     generateSlug(r.name_en, r.constituency_slug),
      dateOfBirth: r.dob || null,
      gender:      r.gender,
      education:   r.education,
      profession:  r.profession,
      isIncumbent: r.is_incumbent === 'true',
      termsServed: r.terms_served,
      criminalCases: r.criminal_cases,
      totalAssetsInr:     r.total_assets_inr || null,
      totalLiabilitiesInr: r.total_liabilities_inr || null,
      affidavitUrl: r.affidavit_url || null,
      nominationStatus: r.nomination_status,
      electionYear: 2026,
    });
  });

  // Only insert if zero errors
  if (errors.length > 0) {
    return { success: false, errors, inserted: 0 };
  }

  await db.insert(candidates).values(toInsert).onConflictDoNothing();

  return { success: true, errors: [], inserted: toInsert.length };
}

function generateSlug(nameEn: string, constituencySlug: string): string {
  const namePart = nameEn.toLowerCase().replace(/[^a-z0-9]+/g, '-').replace(/(^-|-$)/g, '');
  return `${namePart}-${constituencySlug}`;
}

On-Demand Revalidation After Admin Edits

TypeScript — apps/web/app/api/revalidate/route.ts

import { revalidatePath, revalidateTag } from 'next/cache';
import { NextRequest, NextResponse } from 'next/server';

// Called by the API after admin updates a candidate or constituency
export async function POST(req: NextRequest) {
  const secret = req.nextUrl.searchParams.get('secret');

  if (secret !== process.env.REVALIDATION_SECRET) {
    return NextResponse.json({ error: 'Invalid secret' }, { status: 401 });
  }

  const { type, slug } = await req.json();

  switch (type) {
    case 'candidate':
      revalidatePath(`/candidates/${slug}`);
      break;
    case 'constituency':
      revalidatePath(`/constituencies/${slug}`);
      revalidatePath(`/constituencies`);
      break;
    case 'party':
      revalidatePath(`/parties/${slug}`);
      break;
    case 'all':
      revalidatePath('/', 'layout');
      break;
    default:
      return NextResponse.json({ error: 'Unknown type' }, { status: 400 });
  }

  return NextResponse.json({ revalidated: true, type, slug });
}
