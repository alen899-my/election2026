Next.js 14 — App Router

Frontend Pages

Kerala 2026 uses Next.js 14 App Router with a strategic mix of Static Site Generation (SSG), Incremental Static Regeneration (ISR), and Server-Side Rendering (SSR). All 140 constituency pages and 1000+ candidate pages are statically generated at build time for maximum performance and SEO.

Rendering Strategy: Constituency and candidate pages use SSG with generateStaticParams() — built once, served from CDN globally. Pages with frequently changing data (search results, comparison tool) use SSR. Phase 2 live results page uses client-side fetching with WebSocket.

// Generate all 140 constituency pages at build time
export async function generateStaticParams() {
  const slugs = await getAllConstituencySlugs();
  return slugs.map(slug => ({ slug }));
}

// Dynamic metadata for SEO — each constituency gets unique title/description
export async function generateMetadata({ params }: Props): Promise<Metadata> {
  const constituency = await getConstituencyBySlug(params.slug);
  if (!constituency) return {};

  return {
    title: `${constituency.nameEn} Constituency — Kerala 2026 Election`,
    description: `2026 election candidates, historical results, and voter data for ${constituency.nameEn} (${constituency.nameMl}) constituency in ${constituency.district.nameEn} district, Kerala.`,
    openGraph: {
      title: `${constituency.nameEn} — Kerala 2026`,
      images: [`/api/og/constituency/${constituency.slug}`],
    },
    alternates: {
      canonical: `/constituencies/${constituency.slug}`,
    },
  };
}

import Image from 'next/image';
import Link from 'next/link';
import { Candidate } from '@kerala2026/types';
import PartyBadge from '@/components/ui/PartyBadge';
import CriminalCaseTag from '@/components/ui/CriminalCaseTag';
import { formatCrores, calculateAge } from '@kerala2026/utils';
import { cn } from '@/lib/utils';

interface CandidateCardProps {
  candidate:           Candidate;
  variant?:            'default' | 'compact' | 'featured';
  showConstituency?:   boolean;
  showAssets?:         boolean;
  showCriminalBadge?:  boolean;
  priority?:           boolean;
}

export default function CandidateCard({
  candidate,
  variant = 'default',
  showConstituency = false,
  showAssets = false,
  showCriminalBadge = true,
  priority = false,
}: CandidateCardProps) {
  const age = candidate.dateOfBirth
    ? calculateAge(candidate.dateOfBirth)
    : null;

  return (
    <Link
      href={`/candidates/${candidate.slug}`}
      className={cn(
        "group block bg-white border border-border2 rounded-xl overflow-hidden",
        "transition-all duration-200 hover:border-saffron hover:shadow-md hover:-translate-y-1",
        variant === 'compact' && "flex items-center gap-4 p-3",
        variant === 'featured' && "relative",
      )}
    >
      {/* PHOTO */}
      <div className={cn(
        "relative bg-bg2 overflow-hidden",
        variant === 'default' && "aspect-[3/4]",
        variant === 'compact' && "w-16 h-16 rounded-lg flex-shrink-0",
        variant === 'featured' && "aspect-[4/5]",
      )}>
        {candidate.photoUrl ? (
          <Image
            src={candidate.photoUrl}
            alt={`${candidate.nameEn} — ${candidate.party?.nameEn} candidate`}
            fill
            className="object-cover object-top transition-transform duration-300 group-hover:scale-105"
            priority={priority}
            sizes="(max-width: 640px) 50vw, (max-width: 1024px) 33vw, 25vw"
          />
        ) : (
          <div className="w-full h-full flex items-center justify-center text-text3">
            <{/* Placeholder silhouette SVG */}
            <svg className="w-1/3 h-1/3 opacity-20" viewBox="0 0 24 24" fill="currentColor">
              <path d="M12 12c2.7 0 4.8-2.1 4.8-4.8S14.7 2.4 12 2.4 7.2 4.5 7.2 7.2 9.3 12 12 12zm0 2.4c-3.2 0-9.6 1.6-9.6 4.8v2.4h19.2v-2.4c0-3.2-6.4-4.8-9.6-4.8z"/>
            </svg>
          </div>
        )}

        {{/* Incumbent badge */}}
        {candidate.isIncumbent && (
          <div className="absolute top-2 left-2 bg-saffron text-white text-xs font-bold px-2 py-0.5 rounded">
            Incumbent
          </div>
        )}

        {{/* Party color strip at bottom */}}
        if (candidate.party) {
          <div
            className="absolute bottom-0 left-0 right-0 h-1"
            style={{ background: candidate.party.colorHex }}
          />
        }
      </div>

      {{/* INFO */}}
      <div className="p-4">
        {{/* Party badge */}}
        {candidate.party && (
          <PartyBadge party={candidate.party} size="sm" className="mb-2" />
        )}

        {{/* Candidate name in English */}}
        <h3 className="font-display font-bold text-base text-text leading-tight">
          {candidate.nameEn}
        </h3>

        {{/* Candidate name in Malayalam */}}
        {candidate.nameMl && (
          <p className="font-ml text-sm text-text2 mt-0.5">{candidate.nameMl}</p>
        )}

        {showConstituency && candidate.constituency && (
          <p className="text-xs text-text3 mt-1">
            {candidate.constituency.nameEn}
          </p>
        )}

        {{/* Meta row: age, terms */}}
        <div className="flex items-center gap-3 mt-2 text-xs text-text3">
          {age && <span>Age {age}</span>}
          {candidate.termsServed > 0 && (
            <span>{candidate.termsServed} term{candidate.termsServed > 1 ? 's' : ''}</span>
          )}
        </div>

        {{/* Assets + Criminal badge row */}}
        <div className="flex items-center justify-between mt-3">
          {showAssets && candidate.totalAssetsInr && (
            <span className="text-xs text-text2">
              ₹{formatCrores(candidate.totalAssetsInr)}
            </span>
          )}
          {showCriminalBadge && (
            <CriminalCaseTag cases={candidate.criminalCases} />
          )}
        </div>
      </div>
    </Link>
  );
}

import MeiliSearch from 'meilisearch';
import { env } from '../config/env';

export const meilisearch = new MeiliSearch({
  host:   env.MEILISEARCH_HOST,
  apiKey: env.MEILISEARCH_MASTER_KEY,
});

// Configure candidates index
export async function configureCandidatesIndex() {
  const index = meilisearch.index('candidates');

  await index.updateSettings({
    searchableAttributes: [
      'nameEn', 'nameMl', 'party.nameEn', 'party.abbreviation',
      'constituency.nameEn', 'constituency.nameMl', 'profession', 'bio',
    ],
    filterableAttributes: [
      'district', 'party.slug', 'party.coalition',
      'gender', 'isIncumbent', 'electionYear',
      'criminalCases', 'nominationStatus',
    ],
    sortableAttributes: ['nameEn', 'termsServed', 'totalAssetsInr'],
    typoTolerance: {
      enabled: true,
      minWordSizeForTypos: { oneTypo: 4, twoTypos: 8 },
    },
    // Support both Malayalam and English characters
    nonSeparatorTokens: [],
    separatorTokens:    [],
    dictionary:         [],
  });
}

// Sync all candidates to Meilisearch
export async function syncCandidatesToSearch(candidates: any[]) {
  const index = meilisearch.index('candidates');
  await index.addDocuments(candidates, { primaryKey: 'id' });
}

// Search candidates with filters
export async function searchCandidates(query: string, filters?: SearchFilters) {
  const index = meilisearch.index('candidates');
  const filterStr = buildFilterString(filters);

  return index.search(query, {
    limit:             20,
    attributesToHighlight: ['nameEn', 'nameMl'],
    highlightPreTag:   '<mark>',
    highlightPostTag:  '</mark>',
    filter:            filterStr || undefined,
    sort:              filters?.sort || ['nameEn:asc'],
  });
}

SearchBar Component

"use client";

import { useState, useEffect, useRef, useCallback } from 'react';
import { useRouter } from 'next/navigation';
import { useDebouncedCallback } from 'use-debounce';

export default function SearchBar() {
  const [open, setOpen] = useState(false);
  const [query, setQuery] = useState('');
  const [results, setResults] = useState({ candidates: [], constituencies: [], parties: [] });
  const [loading, setLoading] = useState(false);
  const inputRef = useRef<HTMLInputElement>(null);
  const router = useRouter();

  // Keyboard shortcut: ⌘K or Ctrl+K
  useEffect(() => {
    function handler(e: KeyboardEvent) {
      if ((e.metaKey || e.ctrlKey) && e.key === 'k') {
        e.preventDefault();
        setOpen(prev => !prev);
        setTimeout(() => inputRef.current?.focus(), 50);
      }
      if (e.key === 'Escape') setOpen(false);
    }
    document.addEventListener('keydown', handler);
    return () => document.removeEventListener('keydown', handler);
  }, []);

  // Debounced search — fires 300ms after user stops typing
  const search = useDebouncedCallback(async (q: string) => {
    if (!q.trim()) { setResults({ candidates: [], constituencies: [], parties: [] }); return; }
    setLoading(true);
    const res = await fetch(`/api/search?q=${encodeURIComponent(q)}`);
    const data = await res.json();
    setResults(data);
    setLoading(false);
  }, 300);

  return (
    <>
      {{/* Trigger button in header */}}
      <button onClick={() => setOpen(true)}
        className="flex items-center gap-2 px-4 py-2 bg-bg2 rounded-full text-text3 text-sm">
        Search candidates, constituencies...
        <kbd className="ml-auto text-xs bg-white border rounded px-1">⌘K</kbd>
      </button>

      {{/* Modal overlay */}}
      {open && (
        <div className="fixed inset-0 z-50 flex items-start justify-center pt-20 px-4">
          <div className="absolute inset-0 bg-black/50" onClick={() => setOpen(false)} />
          <div className="relative w-full max-w-2xl bg-white rounded-2xl shadow-2xl overflow-hidden">
            <input
              ref={inputRef}
              value={query}
              onChange={e => { setQuery(e.target.value); search(e.target.value); }}
              placeholder="Search candidates, constituencies, parties..."
              className="w-full px-6 py-4 text-lg border-b border-border outline-none"
            />
            <SearchResults results={results} loading={loading} onSelect={() => setOpen(false)} />
          </div>
        </div>
      )}
    </>
  );
}
Data Visualization

Charts & Maps

Kerala 2026 uses Recharts for vote share charts and D3.js for complex custom visualizations. Mapbox GL JS powers all geographic constituency maps. All charts are responsive, animated on mount, and accessible.

VoteShareChart — Full Implementation

TypeScript — components/charts/VoteShareChart.tsx
copy
"use client";

import {
  LineChart, Line, XAxis, YAxis, CartesianGrid, Tooltip,
  Legend, ResponsiveContainer
} from 'recharts';
import { ElectionResult } from '@kerala2026/types';

const PARTY_COLORS = {
  'LDF': '#E31E24',
  'UDF': '#00BFFF',
  'NDA': '#FF6B35',
  'INC': '#00BFFF',
  'CPM': '#E31E24',
  'IND': '#888888',
} as const;

interface Props {
  data:     ElectionResult[];
  height?:  number;
}

// Transform raw results into chart-friendly format
function transformData(data: ElectionResult[]) {
  const byYear = new Map<number, Record<string, number>>();

  data.forEach(result => {
    if (!byYear.has(result.electionYear)) {
      byYear.set(result.electionYear, { year: result.electionYear });
    }
    const yearData = byYear.get(result.electionYear)!;
    yearData[result.party?.coalition || 'IND'] = result.votePercentage;
  });

  return Array.from(byYear.values()).sort((a, b) => a.year - b.year);
}

export default function VoteShareChart({ data, height = 320 }: Props) {
  const chartData = transformData(data);
  const parties = ['LDF', 'UDF', 'NDA'];

  return (
    <ResponsiveContainer width="100%" height={height}>
      <LineChart data={chartData} margin={{ top: 5, right: 20, left: 0, bottom: 5 }}>
        <CartesianGrid strokeDasharray="3 3" stroke="#f0f0f0" />
        <XAxis dataKey="year" tick={{ fontSize: 12, fontFamily: 'var(--fm)' }} />
        <YAxis
          domain={[0, 100]}
          tickFormatter={v => `${v}%`}
          tick={{ fontSize: 12, fontFamily: 'var(--fm)' }}
        />
        <Tooltip
          formatter={(value: number, name: string) => [`${value.toFixed(1)}%`, name]}
          contentStyle={{ fontFamily: 'var(--fb)', fontSize: 12 }}
        />
        <Legend />
        {parties.map(party => (
          <Line
            key={party}
            type="monotone"
            dataKey={party}
            stroke={PARTY_COLORS[party as keyof typeof PARTY_COLORS]}
            strokeWidth={2.5}
            dot={{ r: 4, strokeWidth: 2 }}
            activeDot={{ r: 6 }}
            connectNulls
          />
        ))}
      </LineChart>
    </ResponsiveContainer>
  );
}

Constituency Service Implementation

TypeScript — modules/constituencies/constituencies.service.ts
copy
import { eq, and, isNull, ilike } from 'drizzle-orm';
import { db } from '../../config/db';
import { constituencies, candidates, electionResults, parties, districts } from '@kerala2026/db';

export async function getConstituencyBySlug(slug: string) {
  const [constituency] = await db
    .select()
    .from(constituencies)
    .leftJoin(districts, eq(constituencies.districtId, districts.id))
    .where(and(
      eq(constituencies.slug, slug),
      isNull(constituencies.deletedAt)
    ))
    .limit(1);

  if (!constituency) return null;

  // Fetch candidates for this constituency
  const constituencyCandidates = await db
    .select()
    .from(candidates)
    .leftJoin(parties, eq(candidates.partyId, parties.id))
    .where(and(
      eq(candidates.constituencyId, constituency.constituencies.id),
      eq(candidates.electionYear, 2026),
      isNull(candidates.deletedAt)
    ))
    .orderBy(candidates.isIncumbent); // incumbents first

  // Fetch historical results
  const results = await db
    .select()
    .from(electionResults)
    .leftJoin(parties, eq(electionResults.partyId, parties.id))
    .where(eq(electionResults.constituencyId, constituency.constituencies.id))
    .orderBy(electionResults.electionYear);

  return {
    ...constituency.constituencies,
    district:          constituency.districts,
    candidates:        constituencyCandidates,
    historicalResults: results,
  };
}

Phase 2 — Result Day

Live Results Scraper

On election result day, the Phase 2 scraper activates. It periodically scrapes official results from the Kerala CEO website and the Election Commission of India portal, storing updates in PostgreSQL and broadcasting via Socket.io WebSocket to all connected browsers.

Phase 2 is inactive during Phase 1. The SCRAPER_ENABLED=false environment variable ensures it doesn't run. On result day, flip to true and deploy. The scraper and WebSocket server activate automatically.

import { db } from '../../packages/db';

const CEO_KERALA_URL = 'https://ceo.kerala.gov.in/electoralroll2025.html';
const ECI_RESULTS_URL = 'https://results.eci.gov.in';

export async function scrapeAndUpdate() {
  const browser = await puppeteer.launch({
    headless: true,
    args: ['--no-sandbox', '--disable-setuid-sandbox'],
  });

  try {
    const page = await browser.newPage();

    // Set realistic user agent to avoid blocks
    await page.setUserAgent(
      'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'
    );

    // Navigate to results page
    await page.goto(ECI_RESULTS_URL, {
      waitUntil: 'networkidle2',
      timeout:   30000,
    });

    // Wait for results table to load
    await page.waitForSelector('.result-table', { timeout: 15000 });

    const html = await page.content();
    const $ = cheerio.load(html);
    const updates: LiveResultUpdate[] = [];

    // Parse each constituency row
    $('.result-table tr[data-constituency]').each((_, el) => {
      const constituencyNum = $(el).attr('data-constituency');
      const rows = $(el).find('td');

      updates.push({
        constituencyNumber:  parseInt(constituencyNum!),
        leadingCandidateName: rows.eq(0).text().trim(),
        leadingParty:         rows.eq(1).text().trim(),
        votesSoFar:           parseInt(rows.eq(2).text().replace(/,/g, '')),
        leadMargin:           parseInt(rows.eq(3).text().replace(/,/g, '')),
        roundsCompleted:      parseInt(rows.eq(4).text()),
        status:               rows.eq(5).text().trim() as ResultStatus,
        lastUpdated:          new Date(),
      });
    });

    // Upsert to database
    await upsertLiveResults(updates);

    // Broadcast to all WebSocket clients
    broadcast({ type: 'RESULTS_UPDATE', data: updates, timestamp: new Date() });

    console.log(`Scraped ${updates.length} constituency updates`);
  } catch (err) {
    console.error('Scraping failed:', err);
    // Don't throw — scheduler will retry in 60s
  } finally {
    await browser.close();
  }
}

// Scheduler: runs every 60 seconds on result day
export function startScheduler() {
  if (process.env.SCRAPER_ENABLED !== 'true') {
    console.log('Scraper disabled. Set SCRAPER_ENABLED=true on result day.');
    return;
  }
  console.log('🔴 LIVE SCRAPER ACTIVE — updating every 60 seconds');
  scrapeAndUpdate(); // Run immediately on start
  setInterval(scrapeAndUpdate, 60_000);
}

"use client";

import { useEffect, useState } from 'react';
import { io, Socket } from 'socket.io-client';
import { cn } from '@/lib/utils';

export function useLiveResults() {
  const [results, setResults] = useState<Record<string, LiveResult>>({});
  const [connected, setConnected] = useState(false);

  useEffect(() => {
    const socket: Socket = io(process.env.NEXT_PUBLIC_WEBSOCKET_URL!);

    socket.on('connect', () => setConnected(true));
    socket.on('disconnect', () => setConnected(false));

    socket.on('results:update', (data: ResultsUpdateEvent) => {
      setResults(prev => {
        const updated = { ...prev };
        data.data.forEach(r => { updated[r.constituencySlug] = r; });
        return updated;
      });
    });

    return () => { socket.disconnect(); };
  }, []);

  return { results, connected };
}

export function LiveResultCard({ constituencySlug }: { constituencySlug: string }) {
  const { results, connected } = useLiveResults();
  const result = results[constituencySlug];
  const [pulse, setPulse] = useState(false);

  useEffect(() => {
    if (result) {
      setPulse(true);
      const t = setTimeout(() => setPulse(false), 1000);
      return () => clearTimeout(t);
    }
  }, [result?.votesSoFar]);

  return (
    <div className={cn(
      "p-4 rounded-xl border border-border2 bg-white transition-all",
      pulse && "border-green bg-green-50 shadow-md"
    )}>
      <div className="flex items-center gap-2 mb-3">
        <span className={cn(
          "w-2 h-2 rounded-full",
          connected ? "bg-green animate-pulse" : "bg-gray-300"
        )} />
        <span className="text-xs text-text3">{connected ? 'Live' : 'Connecting...'}</span>
      </div>
      {result ? (
        <>
          <p className="font-display font-bold text-lg">{result.leadingCandidateName}</p>
          <p className="text-sm text-text2">{result.leadingParty}</p>
          <p className="font-mono text-xl font-bold mt-2">
            {result.votesSoFar.toLocaleString('en-IN')}
          </p>
          <p className="text-xs text-text3">
            Lead by {result.leadMargin.toLocaleString('en-IN')} · Round {result.roundsCompleted}
          </p>
        </>
      ) : (
        <p className="text-text3 text-sm">Awaiting results...</p>
      )}
    </div>
  );
}
